20  import com.example.restapi.dto.SaleResponse;
21  import com.example.restapi.service.SaleService;
22  
23  @WebMvcTest(SalesController.class)
24  @DisplayName("SalesController Tests")
25  class SalesControllerTest {
26  
27      private static final Logger log = LoggerFactory.getLogger(SalesControllerTest.class);
28  
29      @Autowired MockMvc mockMvc;
30      @MockitoBean SaleService saleService;
31  
32      @Nested
33      @DisplayName("GET /api/sales/seller/{sellerId}")
34      class GetSalesBySellerIdTests {
35  
36          @Test
37          @DisplayName("should return 200 with sales list for valid seller UUID")
38          void returnsSalesForValidSeller() throws Exception {
39              UUID sellerId = UUID.randomUUID();
40              SaleResponse sale = new SaleResponse(1L, "Orchid", 2, 15.0, 30.0, null);
41              when(saleService.getSalesBySellerId(sellerId)).thenReturn(List.of(sale));
42  
43              mockMvc.perform(get("/api/sales/seller/" + sellerId))
44                      .andExpect(status().isOk())
45                      .andExpect(jsonPath("$[0].itemName").value("Orchid"))
46                      .andExpect(jsonPath("$[0].quantity").value(2));
47  
48              log.info("returnsSalesForValidSeller passed");
49          }
50  
51          @Test
52          @DisplayName("should return 200 with empty list when seller has no sales")
53          void returnsEmptyListForSellerWithNoSales() throws Exception {
54              UUID sellerId = UUID.randomUUID();
55              when(saleService.getSalesBySellerId(sellerId)).thenReturn(List.of());
56  
57              mockMvc.perform(get("/api/sales/seller/" + sellerId))
58                      .andExpect(status().isOk())
59                      .andExpect(jsonPath("$").isEmpty());
60  
61              log.info("returnsEmptyListForSellerWithNoSales passed");
62          }
63  
64          @Test
65          @DisplayName("should return 400 for invalid UUID")
66          void returns400ForInvalidUUID() throws Exception {
67              mockMvc.perform(get("/api/sales/seller/not-a-uuid"))
68                      .andExpect(status().isBadRequest());
69  
70              log.info("returns400ForInvalidUUID passed");
71          }
72      }
73  
74      @Nested
75      @DisplayName("GET /api/sales/seller/{sellerId}/total")
76      class GetTotalSalesBySellerIdTests {
77  
78          @Test
79          @DisplayName("should return 200 with total, count and sales list")
80          void returnsTotalForValidSeller() throws Exception {
81              UUID sellerId = UUID.randomUUID();
82              SaleResponse sale = new SaleResponse(1L, "Fern", 3, 10.0, 30.0, null);
83              when(saleService.getTotalSalesBySellerId(sellerId)).thenReturn(30.0);
84              when(saleService.getSalesBySellerId(sellerId)).thenReturn(List.of(sale));
85  
86              mockMvc.perform(get("/api/sales/seller/" + sellerId + "/total"))
87                      .andExpect(status().isOk())
88                      .andExpect(jsonPath("$.totalSales").value(30.0))
89                      .andExpect(jsonPath("$.numberOfSales").value(1));
90  
91              log.info("returnsTotalForValidSeller passed");
92          }
93  
94          @Test
95          @DisplayName("should return 400 for invalid UUID")
96          void returns400ForInvalidUUID() throws Exception {
97              mockMvc.perform(get("/api/sales/seller/bad-uuid/total"))
98                      .andExpect(status().isBadRequest());
99  
100             log.info("returns400ForInvalidUUID passed");
101         }
