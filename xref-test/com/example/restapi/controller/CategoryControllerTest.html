17  
18  import com.example.restapi.model.Category;
19  import com.example.restapi.repository.CategoryRepository;
20  
21  @WebMvcTest(CategoryController.class)
22  @DisplayName("CategoryController Tests")
23  class CategoryControllerTest {
24  
25      private static final Logger log = LoggerFactory.getLogger(CategoryControllerTest.class);
26  
27      @Autowired MockMvc mockMvc;
28      @MockitoBean CategoryRepository categoryRepository;
29  
30      @Test
31      @DisplayName("GET /api/categories returns 200 with all categories")
32      void returnsAllCategories() throws Exception {
33          Category c1 = new Category("Indoor", "Indoor plants");
34          Category c2 = new Category("Outdoor", "Outdoor plants");
35          when(categoryRepository.findAll()).thenReturn(List.of(c1, c2));
36  
37          mockMvc.perform(get("/api/categories"))
38                  .andExpect(status().isOk())
39                  .andExpect(jsonPath("$.length()").value(2))
40                  .andExpect(jsonPath("$[0].name").value("Indoor"))
41                  .andExpect(jsonPath("$[1].name").value("Outdoor"));
42  
43          log.info("returnsAllCategories passed");
44      }
45  
46      @Test
47      @DisplayName("GET /api/categories returns 200 with empty list when no categories")
48      void returnsEmptyListWhenNoCategories() throws Exception {
49          when(categoryRepository.findAll()).thenReturn(List.of());
50  
51          mockMvc.perform(get("/api/categories"))
52                  .andExpect(status().isOk())
53                  .andExpect(jsonPath("$").isEmpty());
54  
55          log.info("returnsEmptyListWhenNoCategories passed");
56      }
57  }
