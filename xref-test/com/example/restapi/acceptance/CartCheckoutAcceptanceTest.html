15  /**
16   * Acceptance test for the cart + checkout flow. Adds the first available plant to the
17   * cart, opens the cart sidebar, runs the simulated checkout and asserts the receipt.
18   *
19   * PaymentService rejects ~10% of payments at random, so checkout is retried up to
20   * {@link #CHECKOUT_RETRIES} times — leaving the cumulative failure rate negligible.
21   */
22  class CartCheckoutAcceptanceTest extends BaseAcceptanceTest {
23  
24      private static final int CHECKOUT_RETRIES = 4;
25  
26      @Test
27      @DisplayName("user adds an item, checks out and sees the receipt")
28      void addItemAndCheckoutSucceeds() {
29          loginAsTestUser();
30  
31          // 1. Open the first plant and add 1 unit to the cart.
32          wait.until(d -> !d.findElements(By.cssSelector("[data-testid='plant-view-details']")).isEmpty());
33          driver.findElements(By.cssSelector("[data-testid='plant-view-details']")).get(0).click();
34  
35          wait.until(ExpectedConditions.visibilityOfElementLocated(By.cssSelector("[data-testid='plant-details-modal']")));
36          // The modal mounts immediately but renders a spinner while plant details load —
37          // wait for the quantity input itself, which only appears once data has arrived.
38          // The input is pre-filled with 1 via React state, so we leave it untouched
39          // (typing into it cross-platform is unreliable: Ctrl+A doesn't select on macOS).
40          wait.until(ExpectedConditions.visibilityOfElementLocated(By.cssSelector("[data-testid='plant-quantity']")));
41          wait.until(ExpectedConditions.elementToBeClickable(By.cssSelector("[data-testid='plant-add-to-cart']")))
42                  .click();
43  
44          // The modal closes itself after a 1.5s success message. Wait for it to be gone.
45          wait.until(ExpectedConditions.invisibilityOfElementLocated(By.cssSelector("[data-testid='plant-details-modal']")));
46  
47          // 2. Open the cart, expect at least one item.
48          driver.findElement(By.cssSelector("[data-testid='open-cart']")).click();
49          wait.until(ExpectedConditions.visibilityOfElementLocated(By.cssSelector("[data-testid='cart-sidebar']")));
50          wait.until(d -> !d.findElements(By.cssSelector("[data-testid='cart-item']")).isEmpty());
51  
52          assertFalse(driver.findElements(By.cssSelector("[data-testid='cart-item']")).isEmpty(),
53                  "Cart should contain the item we just added");
54  
55          // 3. Run checkout. Retry on the random 10% payment failure.
56          driver.findElement(By.cssSelector("[data-testid='cart-checkout']")).click();
57          wait.until(ExpectedConditions.visibilityOfElementLocated(By.cssSelector("[data-testid='checkout-modal']")));
58  
59          boolean receiptShown = false;
60          for (int attempt = 0; attempt < CHECKOUT_RETRIES && !receiptShown; attempt++) {
61              fillCheckoutForm();
62              driver.findElement(By.cssSelector("[data-testid='checkout-submit']")).click();
63  
64              WebDriverWait outcome = new WebDriverWait(driver, Duration.ofSeconds(20));
65              receiptShown = outcome.until(d -> {
66                  if (!d.findElements(By.cssSelector("[data-testid='receipt-modal']")).isEmpty()) {
67                      return true;
68                  }
69                  // Payment failed (simulated). The form stays open with a .checkout-error message.
70                  return !d.findElements(By.className("checkout-error")).isEmpty();
71              });
72  
73              if (!driver.findElements(By.cssSelector("[data-testid='receipt-modal']")).isEmpty()) {
74                  receiptShown = true;
75              } else {
76                  // Clear the error and retry — the form is still on screen and re-fillable.
77                  receiptShown = false;
78              }
79          }
80  
81          assertTrue(receiptShown,
82                  "Checkout did not produce a receipt after " + CHECKOUT_RETRIES + " attempts");
83  
84          WebElement receiptNumber = wait.until(
85                  ExpectedConditions.visibilityOfElementLocated(By.cssSelector("[data-testid='receipt-number']")));
86          assertFalse(receiptNumber.getText().isBlank(), "Receipt number should be present");
87  
88          driver.findElement(By.cssSelector("[data-testid='receipt-continue']")).click();
89          wait.until(ExpectedConditions.invisibilityOfElementLocated(By.cssSelector("[data-testid='receipt-modal']")));
90      }
91  
92      private void fillCheckoutForm() {
93          setText(By.cssSelector("[data-testid='checkout-card-number']"), "4111111111111111");
94          setText(By.cssSelector("[data-testid='checkout-card-holder']"), "Acceptance Test");
95          setText(By.cssSelector("[data-testid='checkout-expiry']"), "1230");
96          setText(By.cssSelector("[data-testid='checkout-cvv']"), "123");
97      }
98  
99      private void setText(By by, String text) {
100         WebElement el = driver.findElement(by);
101         el.clear();
102         el.sendKeys(text);
103     }
