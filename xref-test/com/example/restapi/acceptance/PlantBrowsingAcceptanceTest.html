14  /**
15   * Acceptance test for the plant catalogue: a logged-in user sees the plants grid and
16   * can open the details modal of any plant. Assumes at least one item exists in the DB
17   * (created via the app or fixtures).
18   */
19  class PlantBrowsingAcceptanceTest extends BaseAcceptanceTest {
20  
21      @Test
22      @DisplayName("logged-in user sees plant cards on the shop tab")
23      void plantsGridShowsCards() {
24          loginAsTestUser();
25  
26          // Shop tab is the default. Wait for the grid to render and then for at least one card.
27          wait.until(ExpectedConditions.visibilityOfElementLocated(By.cssSelector("[data-testid='plants-grid']")));
28          wait.until(d -> !d.findElements(By.cssSelector("[data-testid='plant-card']")).isEmpty());
29  
30          List<WebElement> cards = driver.findElements(By.cssSelector("[data-testid='plant-card']"));
31          assertFalse(cards.isEmpty(),
32                  "Expected at least one plant in the catalogue. "
33                          + "Seed one via the UI ('+ Create post') or directly in the DB before running this test.");
34      }
35  
36      @Test
37      @DisplayName("clicking 'View details' opens the plant details modal")
38      void clickingViewDetailsOpensModal() {
39          loginAsTestUser();
40  
41          wait.until(d -> !d.findElements(By.cssSelector("[data-testid='plant-view-details']")).isEmpty());
42          driver.findElements(By.cssSelector("[data-testid='plant-view-details']")).get(0).click();
43  
44          WebElement modal = wait.until(
45                  ExpectedConditions.visibilityOfElementLocated(By.cssSelector("[data-testid='plant-details-modal']")));
46          assertTrue(modal.isDisplayed(), "Plant details modal should appear after clicking 'View details'");
47          // The add-to-cart button is what makes this modal actionable.
48          wait.until(ExpectedConditions.visibilityOfElementLocated(By.cssSelector("[data-testid='plant-add-to-cart']")));
49      }
50  }
