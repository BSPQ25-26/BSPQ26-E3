20  /**
21   * Base for Selenium acceptance tests. Mirrors dipina/book-api-acceptance-selenium with
22   * one key difference: the stack (Spring Boot + React via nginx) must already be running
23   * — these tests do NOT boot the app via @SpringBootTest because the frontend is a
24   * separate container. Run the stack with `docker compose up -d` before `mvn test -Pacceptance`.
25   */
26  public abstract class BaseAcceptanceTest {
27  
28      protected static final String BASE_URL =
29              System.getProperty("acceptance.baseUrl", "http://localhost");
30  
31      protected static final String TEST_USER_EMAIL =
32              envOrProperty("TEST_USER_EMAIL", "test.user@example.com");
33  
34      protected static final String TEST_USER_PASSWORD =
35              envOrProperty("TEST_USER_PASSWORD", "ChangeMe!123");
36  
37      protected static final Duration DEFAULT_WAIT = Duration.ofSeconds(15);
38  
39      protected WebDriver driver;
40      protected WebDriverWait wait;
41  
42      @BeforeAll
43      static void waitForStack() throws Exception {
44          // Health-check via Spring Boot Actuator. The backend container exposes 8080
45          // directly; nginx (port 80) does NOT proxy /actuator, so we hit 8080.
46          String healthUrl = backendHealthUrl();
47          long deadline = System.currentTimeMillis() + 60_000L;
48          Exception last = null;
49          while (System.currentTimeMillis() < deadline) {
50              try {
51                  HttpURLConnection conn = (HttpURLConnection) URI.create(healthUrl).toURL().openConnection();
52                  conn.setConnectTimeout(2000);
53                  conn.setReadTimeout(2000);
54                  conn.setRequestMethod("GET");
55                  int code = conn.getResponseCode();
56                  conn.disconnect();
57                  if (code == 200) {
58                      return;
59                  }
60              } catch (Exception e) {
61                  last = e;
62              }
63              Thread.sleep(1000);
64          }
65          throw new IllegalStateException(
66                  "Backend not reachable at " + healthUrl
67                          + " — start the stack with `docker compose up -d` before running -Pacceptance",
68                  last);
69      }
70  
71      @BeforeEach
72      void setUpDriver() {
73          WebDriverManager.chromedriver().setup();
74          ChromeOptions options = new ChromeOptions();
75          options.addArguments(
76                  "--headless=new",
77                  "--no-sandbox",
78                  "--disable-dev-shm-usage",
79                  "--window-size=1366,900");
80          driver = new ChromeDriver(options);
81          wait = new WebDriverWait(driver, DEFAULT_WAIT);
82      }
83  
84      @AfterEach
85      void tearDownDriver() {
86          if (driver != null) {
87              driver.quit();
88          }
89      }
90  
91      /** Drives the React login form with the seeded test user and waits for the dashboard. */
92      protected void loginAsTestUser() {
93          driver.get(BASE_URL + "/");
94          wait.until(ExpectedConditions.visibilityOfElementLocated(By.cssSelector("[data-testid='login-email']")))
95                  .sendKeys(TEST_USER_EMAIL);
96          driver.findElement(By.cssSelector("[data-testid='login-password']")).sendKeys(TEST_USER_PASSWORD);
97          driver.findElement(By.cssSelector("[data-testid='login-submit']")).click();
98          wait.until(ExpectedConditions.visibilityOfElementLocated(By.cssSelector("[data-testid='dashboard-shell']")));
99      }
100 
101     /** Convenience: first element matching the selector, or null if none. */
102     protected WebElement firstOrNull(By by) {
103         var elements = driver.findElements(by);
104         return elements.isEmpty() ? null : elements.get(0);
105     }
106 
107     private static String backendHealthUrl() {
108         // BASE_URL is the URL Selenium navigates to (frontend, typically http://localhost).
109         // Health-check always goes to the backend on 8080.
110         return "http://localhost:8080/actuator/health";
111     }
112 
113     private static String envOrProperty(String name, String fallback) {
114         String fromEnv = System.getenv(name);
115         if (fromEnv != null && !fromEnv.isBlank()) {
116             return fromEnv;
117         }
118         String fromProp = System.getProperty(name);
119         if (fromProp != null && !fromProp.isBlank()) {
120             return fromProp;
121         }
122         return fallback;
123     }
124 }
