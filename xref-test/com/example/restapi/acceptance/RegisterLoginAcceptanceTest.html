1   package com.example.restapi.acceptance;

3   import static org.junit.jupiter.api.Assertions.assertTrue;
4   
5   import org.junit.jupiter.api.DisplayName;
6   import org.junit.jupiter.api.Test;
7   import org.openqa.selenium.By;
8   import org.openqa.selenium.WebElement;
9   import org.openqa.selenium.support.ui.ExpectedConditions;
10  
11  /**
12   * Smoke acceptance test: the seeded user (TEST_USER_EMAIL / TEST_USER_PASSWORD) can
13   * sign in through the React UI and reach the dashboard. Verifies the full stack is
14   * wired end-to-end (frontend → nginx → backend → Supabase).
15   */
16  class RegisterLoginAcceptanceTest extends BaseAcceptanceTest {
17  
18      @Test
19      @DisplayName("seeded user logs in and lands on the dashboard")
20      void loginSucceedsAndShowsDashboard() {
21          loginAsTestUser();
22          WebElement dashboard = driver.findElement(By.cssSelector("[data-testid='dashboard-shell']"));
23          assertTrue(dashboard.isDisplayed(), "Dashboard shell should be visible after login");
24      }
25  
26      @Test
27      @DisplayName("invalid credentials surface an error and stay on the login screen")
28      void invalidCredentialsShowError() {
29          driver.get(BASE_URL + "/");
30  
31          WebElement emailInput = wait.until(
32                  ExpectedConditions.visibilityOfElementLocated(By.cssSelector("[data-testid='login-email']")));
33          emailInput.sendKeys("nobody+" + System.currentTimeMillis() + "@example.com");
34          driver.findElement(By.cssSelector("[data-testid='login-password']")).sendKeys("wrong-password");
35          driver.findElement(By.cssSelector("[data-testid='login-submit']")).click();
36  
37          // App routes 404 (unknown email) to the Register screen, and 401 to an inline error.
38          // Both are valid "did not reach the dashboard" outcomes — assert the dashboard is NOT shown.
39          boolean stillOutsideDashboard = wait.until(d ->
40                  d.findElements(By.cssSelector("[data-testid='dashboard-shell']")).isEmpty());
41          assertTrue(stillOutsideDashboard, "Login with bad credentials must not reach the dashboard");
42      }
43  }
