10  /**
11   * Acceptance test for the Purchases and Sales tabs of the dashboard. Verifies that
12   * each tab renders its container without errors — either showing real records or the
13   * empty-state placeholder, both are valid outcomes.
14   */
15  class PurchaseSalesHistoryAcceptanceTest extends BaseAcceptanceTest {
16  
17      @Test
18      @DisplayName("Purchases tab renders the purchase history view")
19      void purchaseTabRenders() {
20          loginAsTestUser();
21  
22          driver.findElement(By.cssSelector("[data-testid='tab-purchases']")).click();
23          wait.until(ExpectedConditions.visibilityOfElementLocated(By.cssSelector("[data-testid='purchase-history']")));
24  
25          boolean rendered =
26                  !driver.findElements(By.cssSelector("[data-testid='purchase-card']")).isEmpty()
27                          || !driver.findElements(By.cssSelector("[data-testid='purchase-history-empty']")).isEmpty();
28          assertTrue(rendered, "Purchase history should render either cards or empty-state placeholder");
29      }
30  
31      @Test
32      @DisplayName("Sales tab renders the sales history view")
33      void salesTabRenders() {
34          loginAsTestUser();
35  
36          driver.findElement(By.cssSelector("[data-testid='tab-sales']")).click();
37          wait.until(ExpectedConditions.visibilityOfElementLocated(By.cssSelector("[data-testid='sales-history']")));
38  
39          boolean rendered =
40                  !driver.findElements(By.cssSelector("[data-testid='sale-card']")).isEmpty()
41                          || !driver.findElements(By.cssSelector("[data-testid='sales-history-empty']")).isEmpty();
42          assertTrue(rendered, "Sales history should render either cards or empty-state placeholder");
43      }
44  }
