113     @Nested
114     @DisplayName("updatePartOfUser")
115     class UpdateUserTests {
116 
117         @Test
118         @DisplayName("should update user successfully")
119         void testUpdateUserSuccess() {
120             Profile updatedProfile = new Profile(testUserId, "newusername", "9876543210");
121             Map<String, Object> updates = new HashMap<>();
122             updates.put("username", "newusername");
123             updates.put("phone", "9876543210");
124 
125             when(profileRepository.findById(testUserId)).thenReturn(Optional.of(testProfile));
126             when(profileRepository.save(any(Profile.class))).thenReturn(updatedProfile);
127 
128             Profile result = userService.updatePartOfUser(testUserId, updates);
129 
130             assertEquals("newusername", result.getUsername());
131             verify(profileRepository).findById(testUserId);
132             verify(profileRepository, atLeastOnce()).save(any(Profile.class));
133             log.info("testUpdateUserSuccess passed: updated username='{}'", result.getUsername());
134         }
135 
136         @Test
137         @DisplayName("should throw exception when user not found")
138         void testUpdateUserNotFound() {
139             Map<String, Object> updates = new HashMap<>();
140             updates.put("username", "newusername");
141             when(profileRepository.findById(testUserId)).thenReturn(Optional.empty());
142 
143             assertThrows(RuntimeException.class, () ->
144                 userService.updatePartOfUser(testUserId, updates)
145             );
146             verify(profileRepository).findById(testUserId);
147             verify(profileRepository, never()).save(any());
148         }
