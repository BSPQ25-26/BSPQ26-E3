42      @Nested
43      @DisplayName("updateOrderStatus")
44      class UpdateOrderStatusTests {
45  
46          @Test
47          @DisplayName("advances PROCESSING orders to DELIVERY when enough time has elapsed")
48          void updateOrderStatus_advancesProcessingToDelivery() {
49              Receipt receipt = new Receipt(UUID.randomUUID(), "REC-1", 50.0, "COMPLETED");
50              receipt.setId(1L);
51              receipt.setProcessingStartedAt(OffsetDateTime.now().minusSeconds(1000));
52  
53              when(receiptRepository.findById(1L)).thenReturn(Optional.of(receipt));
54  
55              boolean progressed = orderStateService.updateOrderStatus(1L);
56  
57              assertTrue(progressed, "Order should progress from PROCESSING to DELIVERY after sufficient elapsed time");
58              assertEquals(OrderStatus.DELIVERY, receipt.getOrderStatus());
59              assertNotNull(receipt.getDeliveryStartedAt());
60              verify(receiptRepository).save(receipt);
61              log.info("updateOrderStatus_advancesProcessingToDelivery passed: newStatus={}", receipt.getOrderStatus());
62          }
63  
64          @Test
65          @DisplayName("advances DELIVERY orders to COMPLETED when enough time has elapsed")
66          void updateOrderStatus_advancesDeliveryToCompleted() {
67              Receipt receipt = new Receipt(UUID.randomUUID(), "REC-2", 50.0, "COMPLETED");
68              receipt.setId(2L);
69              receipt.setOrderStatus(OrderStatus.DELIVERY);
70              receipt.setProcessingStartedAt(OffsetDateTime.now().minusSeconds(1000));
71              receipt.setDeliveryStartedAt(OffsetDateTime.now().minusSeconds(1000));
72  
73              when(receiptRepository.findById(2L)).thenReturn(Optional.of(receipt));
74  
75              boolean progressed = orderStateService.updateOrderStatus(2L);
76  
77              assertTrue(progressed, "Order should progress from DELIVERY to COMPLETED after sufficient elapsed time");
78              assertEquals(OrderStatus.COMPLETED, receipt.getOrderStatus());
79              assertNotNull(receipt.getCompletedAt());
80              verify(receiptRepository).save(receipt);
81              log.info("updateOrderStatus_advancesDeliveryToCompleted passed: newStatus={}", receipt.getOrderStatus());
82          }
83  
84          @Test
85          @DisplayName("does not advance state when not enough time has elapsed")
86          void updateOrderStatus_doesNotAdvanceBeforeTimeElapsed() {
87              Receipt receipt = new Receipt(UUID.randomUUID(), "REC-3", 50.0, "COMPLETED");
88              receipt.setId(3L);
89              receipt.setProcessingStartedAt(OffsetDateTime.now());
90  
91              when(receiptRepository.findById(3L)).thenReturn(Optional.of(receipt));
92  
93              boolean progressed = orderStateService.updateOrderStatus(3L);
94  
95              assertFalse(progressed, "Order should not advance before enough time has elapsed");
96              assertEquals(OrderStatus.PROCESSING, receipt.getOrderStatus());
97          }
98      }
