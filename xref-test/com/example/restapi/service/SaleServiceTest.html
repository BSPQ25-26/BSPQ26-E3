46  
47          @Test
48          @DisplayName("should create one Sale per cart item")
49          void createsOneSalePerCartItem() {
50              UUID sellerId = UUID.randomUUID();
51              Profile seller = new Profile(sellerId, "seller", "000");
52  
53              Item item1 = new Item("Rose", "Red rose", 5.0, null, 10, null, true);
54              item1.setSeller(seller);
55              Item item2 = new Item("Cactus", "Small cactus", 8.0, null, 5, null, true);
56              item2.setSeller(seller);
57  
58              CartItem ci1 = new CartItem(item1, 2);
59              CartItem ci2 = new CartItem(item2, 1);
60  
61              Cart cart = new Cart(UUID.randomUUID());
62              cart.addItem(ci1);
63              cart.addItem(ci2);
64  
65              Receipt receipt = new Receipt();
66  
67              saleService.recordSalesFromCheckout(receipt, cart);
68  
69              verify(saleRepository, times(2)).save(any(Sale.class));
70              log.info("createsOneSalePerCartItem passed: 2 sales saved");
71          }
72  
73          @Test

86      @Nested
87      @DisplayName("getSalesBySellerId")
88      class GetSalesBySellerIdTests {
89  
90          @Test
91          @DisplayName("should return mapped SaleResponse list for seller")
92          void returnsMappedSaleResponses() {
93              UUID sellerId = UUID.randomUUID();
94  
95              Profile seller = new Profile(sellerId, "seller", "000");
96              Item item = new Item("Orchid", "Pretty flower", 15.0, null, 3, null, true);
97              item.setSeller(seller);
98  
99              Sale sale = new Sale(sellerId, new Receipt(), item, 2, 15.0);
100             sale.setId(1L);
101 
102             when(saleRepository.findBySellerId(sellerId)).thenReturn(List.of(sale));
103 
104             List<SaleResponse> result = saleService.getSalesBySellerId(sellerId);
105 
106             assertEquals(1, result.size());
107             assertEquals("Orchid", result.get(0).getItemName());
108             assertEquals(2, result.get(0).getQuantity());
109             assertEquals(15.0, result.get(0).getUnitPrice());
110             assertEquals(30.0, result.get(0).getTotalPrice());
111             log.info("returnsMappedSaleResponses passed: sellerId={}", sellerId);
112         }
113
