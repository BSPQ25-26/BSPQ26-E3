31  @DisplayName("Review Integration Tests")
32  class ReviewServiceIntegrationTest {

34      private static final Logger log = LoggerFactory.getLogger(ReviewServiceIntegrationTest.class);
35  
36      @Autowired private TestRestTemplate restTemplate;
37      @Autowired private ReviewRepository reviewRepository;
38      @Autowired private ItemRepository itemRepository;
39      @Autowired private ProfileRepository profileRepository;
40      @Autowired private CategoryRepository categoryRepository;
41  
42      private Long itemId;
43      private UUID reviewerId;
44  
45      @BeforeEach
46      void setUp() {
47          reviewRepository.deleteAll();
48          itemRepository.deleteAll();
49          profileRepository.deleteAll();
50          categoryRepository.deleteAll();
51  
52          Category category = categoryRepository.save(new Category("Succulents", "Succulent plants"));
53          Profile seller = profileRepository.save(new Profile(UUID.randomUUID(), "item-seller", "000000001"));
54          Profile reviewer = profileRepository.save(new Profile(UUID.randomUUID(), "reviewer-user", "000000002"));
55          reviewerId = reviewer.getId();
56  
57          Item item = new Item("Aloe Vera", "Medicinal plant", 12.50, null, 20, category, true);
58          item.setSeller(seller);
59          itemId = itemRepository.save(item).getId();
60  
61          log.info("setUp complete: itemId={}, reviewerId={}", itemId, reviewerId);
62      }
63  
64      @AfterEach
65      void tearDown() {
66          reviewRepository.deleteAll();
67          itemRepository.deleteAll();
68          profileRepository.deleteAll();
69          categoryRepository.deleteAll();
70      }

72      @Test
73      @DisplayName("POST /api/items/{itemId}/reviews creates a review and returns 201")
74      void createReviewReturns201() {
75          ReviewRequest request = new ReviewRequest();
76          request.setAuthorId(reviewerId);
77          request.setRating(4);
78          request.setComment("Great plant!");
79  
80          ResponseEntity<ReviewResponse> response = restTemplate.postForEntity(
81                  "/api/items/" + itemId + "/reviews", request, ReviewResponse.class);
82  
83          assertEquals(201, response.getStatusCode().value());
84          ReviewResponse body = response.getBody();
85          assertNotNull(body);
86          assertNotNull(body.getId());
87          assertEquals(4, body.getRating());
88          assertEquals("Great plant!", body.getComment());
89          assertEquals(reviewerId, body.getAuthorId());
90          log.info("createReviewReturns201 passed: reviewId={}", body.getId());
91      }
92  
93      @Test
94      @DisplayName("GET /api/items/{itemId}/reviews returns all reviews for the item")
95      void getReviewsByItemReturnsCreatedReview() {
96          ReviewRequest request = new ReviewRequest();
97          request.setAuthorId(reviewerId);
98          request.setRating(5);
99          request.setComment("Excellent!");
100         restTemplate.postForEntity("/api/items/" + itemId + "/reviews", request, ReviewResponse.class);
101 
102         ResponseEntity<ReviewResponse[]> response = restTemplate.getForEntity(
103                 "/api/items/" + itemId + "/reviews", ReviewResponse[].class);
104 
105         assertEquals(200, response.getStatusCode().value());
106         ReviewResponse[] body = response.getBody();
107         assertNotNull(body);
108         assertEquals(1, body.length);
109         assertEquals(5, body[0].getRating());
110         assertEquals("Excellent!", body[0].getComment());
111         log.info("getReviewsByItemReturnsCreatedReview passed: {} review(s) found", body.length);
112     }

126     @Test
127     @DisplayName("DELETE /api/items/{itemId}/reviews/{reviewId} removes the review")
128     void deleteReviewRemovesIt() {
129         ReviewRequest request = new ReviewRequest();
130         request.setAuthorId(reviewerId);
131         request.setRating(3);
132         request.setComment("Decent plant");
133         Long reviewId = restTemplate.postForEntity(
134                 "/api/items/" + itemId + "/reviews", request, ReviewResponse.class)
135                 .getBody().getId();
136 
137         restTemplate.delete("/api/items/" + itemId + "/reviews/" + reviewId + "?requesterId=" + reviewerId);
138 
139         ResponseEntity<ReviewResponse[]> listResponse = restTemplate.getForEntity(
140                 "/api/items/" + itemId + "/reviews", ReviewResponse[].class);
141         assertEquals(200, listResponse.getStatusCode().value());
142         assertEquals(0, listResponse.getBody().length);
143         log.info("deleteReviewRemovesIt passed: reviewId={}", reviewId);
144     }
