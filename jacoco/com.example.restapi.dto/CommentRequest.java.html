import java.util.UUID;

    private Long itemId;
    private Long postId;

        return content;
    }

    public void setContent(String content) {
        this.content = content;

        return itemId;
    }

    public void setItemId(Long itemId) {
        this.itemId = itemId;
    }

    public Long getPostId() {
        return postId;
    }

    public void setPostId(Long postId) {
        this.postId = postId;
    }
}
