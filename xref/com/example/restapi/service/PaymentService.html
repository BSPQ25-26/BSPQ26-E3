1   package com.example.restapi.service;
2   
3   import org.springframework.stereotype.Service;
4   
5   import com.example.restapi.dto.PaymentRequest;
6   
7   /**
8    *
9    * Service layer that simulates payment processing.
10   *
11   * Validates card data (number, expiry, CVV, holder name) and returns a
12   * probabilistic success result.  This remote facade allows the checkout
13   * flow to remain agnostic of the real payment provider.
14   */
15  @Service
16  public class PaymentService {
17  
18      /**
19       * Validates payment details and simulates a charge.
20       *
21       * The simulation has a 90 % success rate when all fields are valid.
22       *
23       * @param paymentRequest DTO containing card holder, number, expiry and CVV.
24       * @param amount         Monetary amount to charge (must be > 0).
25       * @return true if the simulated payment succeeds, false otherwise.
26       */
27      public boolean processPayment(PaymentRequest paymentRequest, Double amount) {
28          if (paymentRequest == null || amount == null || amount <= 0) {
29              return false;
30          }
31  
32          if (!isValidCardNumber(paymentRequest.getCardNumber())) {
33              return false;
34          }
35          if (!isValidExpiryDate(paymentRequest.getExpiryDate())) {
36              return false;
37          }
38          if (!isValidCVV(paymentRequest.getCvv())) {
39              return false;
40          }
41          if (paymentRequest.getCardHolder() == null || paymentRequest.getCardHolder().trim().isEmpty()) {
42              return false;
43          }
44  
45          // Simulate successful payment (90% success rate for testing)
46          return Math.random() < 0.9;
47      }
48  
49      /**
50       * Checks whether a card number looks valid.
51       * @param cardNumber Raw card number (may contain spaces or dashes).
52       * @return true if the cleaned number has 13-19 digits.
53       */
54      private boolean isValidCardNumber(String cardNumber) {
55          if (cardNumber == null || cardNumber.isEmpty()) {
56              return false;
57          }
58          String cleaned = cardNumber.replaceAll("[\\s-]", "");
59          return cleaned.matches("\\d{13,19}");
60      }
61  
62      /**
63       * Validates the expiry date format.
64       * @param expiryDate String expected in MM/YY format.
65       * @return true if the format matches MM/YY.
66       */
67      private boolean isValidExpiryDate(String expiryDate) {
68          if (expiryDate == null || expiryDate.isEmpty()) {
69              return false;
70          }
71          return expiryDate.matches("^(0[1-9]|1[0-2])/\\d{2}$");
72      }
73  
74      /**
75       * Validates the CVV/CVC code.
76       * @param cvv Security code.
77       * @return true if the CVV consists of 3 or 4 digits.
78       */
79      private boolean isValidCVV(String cvv) {
80          if (cvv == null || cvv.isEmpty()) {
81              return false;
82          }
83          return cvv.matches("\\d{3,4}");
84      }
85  }
