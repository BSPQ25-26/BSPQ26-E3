1   package com.example.restapi.service;

3   import com.example.restapi.dto.PostResponse;
4   import com.example.restapi.model.Category;
5   import com.example.restapi.model.Post;
6   import com.example.restapi.repository.PostRepository;
7   import org.springdoc.api.OpenApiResourceNotFoundException;
8   import org.springframework.beans.factory.annotation.Autowired;
9   import org.springframework.stereotype.Service;
10  import org.springframework.transaction.annotation.Transactional;
11  
12  import jakarta.persistence.EntityManager;
13  
14  import java.util.List;
15  import java.util.Map;
16  import java.util.UUID;
17  import java.util.stream.Collectors;
18  
19  /**
20   *
21   * Service layer for blog-post management.
22   *
23   * Provides CRUD operations for posts, author resolution and category
24   * linkage.  This remote facade is consumed by the REST controllers.
25   */
26  @Service
27  public class PostService {
28  
29      private final PostRepository postRepository;
30      private final EntityManager entityManager;
31  
32      /**
33       * Constructs PostService.
34       * @param postRepository Data-access object for Post entities.
35       * @param entityManager  JPA EntityManager used for flush/refresh operations.
36       */
37      @Autowired
38      public PostService(PostRepository postRepository, EntityManager entityManager) {
39          this.postRepository = postRepository;
40          this.entityManager = entityManager;
41      }
42  
43      /**
44       * Retrieves a single post by id.
45       * @param id Post primary key.
46       * @return PostResponse DTO.
47       * @throws OpenApiResourceNotFoundException if the post does not exist.
48       */
49      @Transactional(readOnly = true)
50      public PostResponse getPostById(Long id) {
51          Post post = postRepository.findById(id)
52                  .orElseThrow(() -> new OpenApiResourceNotFoundException("Post id : " + id));
53          return PostResponse.fromEntity(post);
54      }
55  
56      /**
57       * Returns all posts ordered by creation date (newest first).
58       * @return List of PostResponse DTOs.
59       */
60      @Transactional(readOnly = true)
61      public List<PostResponse> getAllPosts() {
62          return postRepository.findAllByOrderByCreatedAtDescIdDesc().stream()
63                  .map(PostResponse::fromEntity)
64                  .collect(Collectors.toList());
65      }
66  
67      /**
68       * Returns every post written by a specific author.
69       * @param authorId UUID of the author.
70       * @return List of PostResponse DTOs.
71       */
72      @Transactional(readOnly = true)
73      public List<PostResponse> getPostsByAuthor(UUID authorId) {
74          return postRepository.findByAuthor_Id(authorId).stream()
75                  .map(PostResponse::fromEntity)
76                  .collect(Collectors.toList());
77      }
78  
79      /**
80       * Persists a new post.
81       *
82       * Flushes and refreshes the entity so that the returned DTO contains
83       * fully loaded author and category names.
84       *
85       * @param post The post to create (title must not be blank).
86       * @return PostResponse DTO of the saved post.
87       * @throws IllegalArgumentException if the title is missing or blank.
88       */
89      @Transactional
90      public PostResponse createPost(Post post) {
91          if (post.getTitle() == null || post.getTitle().isBlank()) {
92              throw new IllegalArgumentException("Post title is required");
93          }
94          Post saved = postRepository.save(post);
95          entityManager.flush();
96          entityManager.refresh(saved);
97          return PostResponse.fromEntity(saved);
98      }
99  
100     /**
101      * Deletes a post.
102      * @param id Post primary key.
103      * @throws OpenApiResourceNotFoundException if the post does not exist.
104      */
105     @Transactional
106     public void deletePost(Long id) {
107         postRepository.findById(id)
108                 .orElseThrow(() -> new OpenApiResourceNotFoundException("Post id : " + id));
109         postRepository.deleteById(id);
110     }
111 
112     /**
113      * Applies a partial update to an existing post.
114      * @param id      Post primary key.
115      * @param updates Map of fields to update (supports "title", "content", "categoryId").
116      * @return PostResponse DTO of the updated post.
117      * @throws RuntimeException if the post is not found.
118      */
119     @Transactional
120     public PostResponse updatePost(Long id, Map<String, Object> updates) {
121         Post post = postRepository.findById(id)
122                 .orElseThrow(() -> new RuntimeException("Post not found with id: " + id));
123         partialUpdate(post, updates);
124         Post saved = postRepository.save(post);
125         if (updates.containsKey("categoryId")) {
126             entityManager.flush();
127             entityManager.refresh(saved);
128         }
129         return PostResponse.fromEntity(saved);
130     }
131 
132     private void partialUpdate(Post postDetails, Map<String, Object> updates) {
133         if (updates.containsKey("title")) {
134             postDetails.setTitle((String) updates.get("title"));
135         }
136 
137         if (updates.containsKey("content")) {
138             postDetails.setContent((String) updates.get("content"));
139         }
140 
141         if (updates.containsKey("categoryId")) {
142             Object raw = updates.get("categoryId");
143             Long categoryId = raw instanceof Number n ? n.longValue() : Long.valueOf(raw.toString());
144             Category category = new Category();
145             category.setId(categoryId);
146             postDetails.setCategory(category);
147         }
148     }
149 }
