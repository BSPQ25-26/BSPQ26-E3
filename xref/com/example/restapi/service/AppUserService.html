59      /**
60       * Registers a new user in Supabase Auth and creates a local profile.
61       * @param req Registration data (email, password, username, phone).
62       * @return The persisted local Profile.
63       * @throws RuntimeException if Supabase signup fails or the username is already taken.
64       */
65      public Profile register(RegisterRequest req) {
66          HttpHeaders headers = new HttpHeaders();
67          headers.set("apikey", supabaseAnonKey);
68          headers.setContentType(MediaType.APPLICATION_JSON);
69  
70          Map<String, Object> body = new HashMap<>();
71          body.put("email", req.getEmail());
72          body.put("password", req.getPassword());
73  
74          HttpEntity<Map<String, Object>> request = new HttpEntity<>(body, headers);
75  
76          Map<String, Object> response;
77          try {
78              @SuppressWarnings("unchecked")
79              Map<String, Object> r = restTemplate.postForObject(
80                      supabaseUrl + "/auth/v1/signup", request, Map.class);
81              response = r;
82          } catch (HttpClientErrorException e) {
83              log.error("Supabase signup error {} - {}", e.getStatusCode(), e.getResponseBodyAsString());
84              throw new RuntimeException("Supabase signup failed: " + e.getResponseBodyAsString());
85          }
86  
87          log.info("Supabase signup response: {}", response);
88          String userId = extractUserId(response);
89  
90          if (profileRepository.findByUsername(req.getUsername()).isPresent()) {
91              throw new RuntimeException("Username already taken: " + req.getUsername());
92          }
93  
94          Profile profile = new Profile(UUID.fromString(userId), req.getUsername(), req.getPhone());
95          return profileRepository.save(profile);
96      }

98      /**
99       * Authenticates a user against Supabase and returns session data.
100      * @param email    User email.
101      * @param password User password.
102      * @return AuthResponse containing the access token and local profile.
103      * @throws RuntimeException if credentials are invalid or the email is not confirmed.
104      */
105     public AuthResponse login(String email, String password) {
106         HttpHeaders headers = new HttpHeaders();
107         headers.set("apikey", supabaseAnonKey);
108         headers.setContentType(MediaType.APPLICATION_JSON);
109 
110         Map<String, String> body = new HashMap<>();
111         body.put("email", email);
112         body.put("password", password);
113 
114         HttpEntity<Map<String, String>> request = new HttpEntity<>(body, headers);
115 
116         Map<String, Object> response;
117         try {
118             @SuppressWarnings("unchecked")
119             Map<String, Object> r = restTemplate.postForObject(
120                     supabaseUrl + "/auth/v1/token?grant_type=password", request, Map.class);
121             response = r;
122         } catch (HttpClientErrorException e) {
123             log.error("Supabase login error {} - {}", e.getStatusCode(), e.getResponseBodyAsString());
124             String errorBody = e.getResponseBodyAsString();
125             if (errorBody.contains("email_not_confirmed")) {
126                 throw new RuntimeException("EMAIL_NOT_CONFIRMED");
127             }
128             throw new RuntimeException("Supabase login failed: " + errorBody);
129         }
130 
131         String accessToken = (String) response.get("access_token");
132         String tokenType = (String) response.get("token_type");
133         Object expiresInRaw = response.get("expires_in");
134         int expiresIn = expiresInRaw instanceof Number ? ((Number) expiresInRaw).intValue() : 3600;
135 
136         @SuppressWarnings("unchecked")
137         Map<String, Object> userMap = (Map<String, Object>) response.get("user");
138         UUID userId = UUID.fromString((String) userMap.get("id"));
139         String userEmail = (String) userMap.get("email");
140 
141         Profile profile = profileRepository.findById(userId)
142                 .orElseThrow(() -> new RuntimeException("Profile not found for user: " + userId));
143 
144         AuthResponse authResponse = new AuthResponse();
145         authResponse.setAccessToken(accessToken);
146         authResponse.setTokenType(tokenType);
147         authResponse.setExpiresIn(expiresIn);
148         authResponse.setEmail(userEmail);
149         authResponse.setProfile(profile);
150         return authResponse;
151     }

176     /**
177      * Retrieves a public profile by username.
178      * @param email    Optional email filter (currently unused).
179      * @param username The unique username to look up.
180      * @return Optional containing the public UserProfileResponse.
181      */
182     public Optional<UserProfileResponse> getDisplayProfile(String email, String username) {
183         Optional<Profile> profileOpt = Optional.empty();
184 
185         if (username != null && !username.isBlank()) {
186             profileOpt = profileRepository.findByUsername(username);
187         }
188 
189         return profileOpt.map(profile -> {
190             UserProfileResponse dto = new UserProfileResponse();
191             dto.setUsername(profile.getUsername());
192             dto.setPhone(profile.getPhone());
193             dto.setCreatedAt(profile.getCreatedAt());
194             return dto;
195         });
196     }

215     /**
216      * Applies a partial update to an existing profile.
217      * @param id      UUID of the profile to update.
218      * @param updates Map of field names to new values (supports "username", "phone").
219      * @return The updated Profile entity.
220      * @throws RuntimeException if the profile does not exist.
221      */
222     public Profile updatePartOfUser(UUID id, Map<String, Object> updates) {
223         Profile profile = profileRepository.findById(id)
224                 .orElseThrow(() -> new RuntimeException("Profile not found with id: " + id));
225         partialUpdate(profile, updates);
226         return profileRepository.save(profile);
227     }

229     private void partialUpdate(Profile userDetails, Map<String, Object> updates){
230         if(updates.containsKey("username")){
231             userDetails.setUsername((String) updates.get("username"));
232         }
233 
234         if(updates.containsKey("phone")){
235             userDetails.setPhone((String) updates.get("phone"));
236         }
237         profileRepository.save(userDetails);
238     }

262     /**
263      * Changes a user's password after verifying the current one.
264      * @param userId          UUID of the user.
265      * @param email           User email.
266      * @param currentPassword The existing password.
267      * @param newPassword     The desired new password.
268      * @throws RuntimeException if the current password is invalid or the update fails.
269      */
270     public void changePassword(UUID userId, String email, String currentPassword, String newPassword) {
271         HttpHeaders verifyHeaders = new HttpHeaders();
272         verifyHeaders.set("apikey", supabaseAnonKey);
273         verifyHeaders.setContentType(MediaType.APPLICATION_JSON);
274 
275         Map<String, String> loginBody = new HashMap<>();
276         loginBody.put("email", email);
277         loginBody.put("password", currentPassword);
278 
279         HttpEntity<Map<String, String>> loginRequest = new HttpEntity<>(loginBody, verifyHeaders);
280         try {
281             restTemplate.postForObject(
282                     supabaseUrl + "/auth/v1/token?grant_type=password", loginRequest, Map.class);
283         } catch (HttpClientErrorException e) {
284             throw new RuntimeException("INVALID_CURRENT_PASSWORD");
285         }
286 
287         HttpHeaders adminHeaders = new HttpHeaders();
288         adminHeaders.set("apikey", supabaseServiceRoleKey);
289         adminHeaders.set("Authorization", "Bearer " + supabaseServiceRoleKey);
290         adminHeaders.setContentType(MediaType.APPLICATION_JSON);
291 
292         Map<String, String> updateBody = new HashMap<>();
293         updateBody.put("password", newPassword);
294 
295         HttpEntity<Map<String, String>> updateRequest = new HttpEntity<>(updateBody, adminHeaders);
296         try {
297             restTemplate.exchange(
298                     supabaseUrl + "/auth/v1/admin/users/" + userId,
299                     HttpMethod.PUT,
300                     updateRequest,
301                     Map.class);
302         } catch (HttpClientErrorException e) {
303             log.error("Supabase password update error {} - {}", e.getStatusCode(), e.getResponseBodyAsString());
304             throw new RuntimeException("Failed to update password: " + e.getResponseBodyAsString());
305         }
306     }

308     /**
309      * Deletes a user both from Supabase Auth and the local database.
310      * @param id UUID of the user to remove.
311      * @throws RuntimeException if the local profile is not found.
312      */
313     public void deleteUser(UUID id) {
314         if (!profileRepository.existsById(id)) {
315             throw new RuntimeException("Profile not found with id: " + id);
316         }
317 
318         HttpHeaders headers = new HttpHeaders();
319         headers.set("apikey", supabaseServiceRoleKey);
320         headers.set("Authorization", "Bearer " + supabaseServiceRoleKey);
321 
322         HttpEntity<Void> request = new HttpEntity<>(headers);
323         restTemplate.exchange(
324                 supabaseUrl + "/auth/v1/admin/users/" + id,
325                 HttpMethod.DELETE,
326                 request,
327                 Void.class);
328         // The profile is deleted automatically by the ON DELETE CASCADE constraint
329     }

331     @SuppressWarnings("unchecked")
332     private String extractUserId(Map<String, Object> response) {
333         if (response.containsKey("user") && response.get("user") != null) {
334             Map<String, Object> user = (Map<String, Object>) response.get("user");
335             return (String) user.get("id");
336         }
337         return (String) response.get("id");
338     }
