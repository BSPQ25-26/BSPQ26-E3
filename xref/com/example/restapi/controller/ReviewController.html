1   package com.example.restapi.controller;
2   
3   import java.util.List;
4   import java.util.UUID;
5   
6   import org.springframework.http.HttpStatus;
7   import org.springframework.http.ResponseEntity;
8   import org.springframework.web.bind.annotation.DeleteMapping;
9   import org.springframework.web.bind.annotation.GetMapping;
10  import org.springframework.web.bind.annotation.PathVariable;
11  import org.springframework.web.bind.annotation.PostMapping;
12  import org.springframework.web.bind.annotation.RequestBody;
13  import org.springframework.web.bind.annotation.RequestMapping;
14  import org.springframework.web.bind.annotation.RequestParam;
15  import org.springframework.web.bind.annotation.RestController;
16  
17  import com.example.restapi.dto.ReviewRequest;
18  import com.example.restapi.dto.ReviewResponse;
19  import com.example.restapi.service.ReviewService;
20  
21  @RestController
22  @RequestMapping("/api/items/{itemId}/reviews")
23  public class ReviewController {
24  
25      private final ReviewService reviewService;
26  
27      public ReviewController(ReviewService reviewService) {
28          this.reviewService = reviewService;
29      }
30  
31      @GetMapping
32      public ResponseEntity<List<ReviewResponse>> getReviewsByItemId(@PathVariable Long itemId) {
33          return ResponseEntity.ok(reviewService.getReviewsByItemId(itemId));
34      }
35  
36      @PostMapping
37      public ResponseEntity<ReviewResponse> createReview(@PathVariable Long itemId, @RequestBody ReviewRequest request) {
38          ReviewResponse created = reviewService.createReview(itemId, request);
39          return new ResponseEntity<>(created, HttpStatus.CREATED);
40      }
41  
42      @DeleteMapping("/{reviewId}")
43      public ResponseEntity<Void> deleteReview(
44              @PathVariable Long itemId,
45              @PathVariable Long reviewId,
46              @RequestParam UUID requesterId) {
47          reviewService.deleteReview(reviewId, requesterId);
48          return ResponseEntity.noContent().build();
49      }
50  }
