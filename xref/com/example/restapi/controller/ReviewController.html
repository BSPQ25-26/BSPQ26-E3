3   import java.util.List;

5   import org.springframework.http.HttpStatus;

7   import org.springframework.web.bind.annotation.GetMapping;

10  import org.springframework.web.bind.annotation.RequestBody;
11  import org.springframework.web.bind.annotation.RequestMapping;
12  import org.springframework.web.bind.annotation.RestController;
13  
14  import com.example.restapi.dto.ReviewRequest;
15  import com.example.restapi.dto.ReviewResponse;
16  import com.example.restapi.service.ReviewService;
17  
18  @RestController
19  @RequestMapping("/api/items/{itemId}/reviews")
20  public class ReviewController {
21  
22      private final ReviewService reviewService;
23  
24      public ReviewController(ReviewService reviewService) {
25          this.reviewService = reviewService;
26      }
27  
28      @GetMapping
29      public ResponseEntity<List<ReviewResponse>> getReviewsByItemId(@PathVariable Long itemId) {
30          return ResponseEntity.ok(reviewService.getReviewsByItemId(itemId));
31      }
32  
33      @PostMapping
34      public ResponseEntity<ReviewResponse> createReview(@PathVariable Long itemId, @RequestBody ReviewRequest request) {
35          ReviewResponse created = reviewService.createReview(itemId, request);
36          return new ResponseEntity<>(created, HttpStatus.CREATED);
37      }
38  }
