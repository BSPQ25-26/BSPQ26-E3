1   package com.example.restapi.controller;
2   
3   import java.util.List;
4   import java.util.UUID;
5   import org.springframework.http.ResponseEntity;
6   import org.springframework.web.bind.annotation.*;
7   import com.example.restapi.dto.ItemResponse;
8   import com.example.restapi.model.Item;
9   import com.example.restapi.service.ItemService;
10  import io.swagger.v3.oas.annotations.tags.Tag;
11  
12  @RestController
13  @RequestMapping("/api/items")
14  @Tag(name = "Item Controller", description = "API for managing items/plants")
15  public class ItemController {
16  
17      private final ItemService itemService;
18  
19      public ItemController(ItemService itemService) {
20          this.itemService = itemService;
21      }
22  
23      @GetMapping
24      public ResponseEntity<List<ItemResponse>> getAllItems() {
25          return ResponseEntity.ok(itemService.getAllItems());
26      }
27  
28      @GetMapping("/{id}")
29      public ResponseEntity<ItemResponse> getItemById(@PathVariable Long id) {
30          try {
31              return ResponseEntity.ok(itemService.getItemById(id));
32          } catch (RuntimeException e) {
33              return ResponseEntity.notFound().build();
34          }
35      }
36  
37      @GetMapping("/active")
38      public ResponseEntity<List<ItemResponse>> getActiveItems() {
39          return ResponseEntity.ok(itemService.getActiveItems());
40      }
41  
42      @PostMapping
43      public ResponseEntity<Item> createItem(@RequestBody Item item, @RequestHeader(value = "X-User-Id", required = false) String userId) {
44          try {
45              System.out.println("DEBUG: Received userId: " + userId);  // ← Agrega
46              UUID sellerId;
47              if (userId != null && !userId.isEmpty()) {
48                  sellerId = UUID.fromString(userId);
49              } else {
50                  return ResponseEntity.badRequest().build();
51              }
52              System.out.println("DEBUG: Creating item with sellerId: " + sellerId);  // ← Agrega
53              return ResponseEntity.ok(itemService.createItem(item, sellerId));
54          } catch (IllegalArgumentException e) {
55              System.err.println("ERROR (BadRequest): " + e.getMessage());  // ← Agrega
56              e.printStackTrace();
57              return ResponseEntity.badRequest().build();
58          } catch (RuntimeException e) {
59              System.err.println("ERROR (500): " + e.getMessage());  // ← Agrega
60              e.printStackTrace();
61              return ResponseEntity.status(500).build();
62          }
63      }
64  
65      @PutMapping("/{id}")
66      public ResponseEntity<Item> updateItem(@PathVariable Long id, @RequestBody Item itemDetails) {
67          try {
68              return ResponseEntity.ok(itemService.updateItem(id, itemDetails));
69          } catch (RuntimeException e) {
70              return ResponseEntity.notFound().build();
71          }
72      }
73  
74      @DeleteMapping("/{id}")
75      public ResponseEntity<Void> deleteItem(@PathVariable Long id) {
76          try {
77              itemService.deleteItem(id);
78              return ResponseEntity.noContent().build();
79          } catch (RuntimeException e) {
80              return ResponseEntity.notFound().build();
81          }
82      }
83  }
