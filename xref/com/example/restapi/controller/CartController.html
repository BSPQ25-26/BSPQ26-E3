1   package com.example.restapi.controller;
2   
3   import java.util.UUID;
4   
5   import org.springframework.http.ResponseEntity;
6   import org.springframework.web.bind.annotation.*;
7   
8   import com.example.restapi.dto.CartRequest;
9   import com.example.restapi.dto.CartResponse;
10  import com.example.restapi.dto.PaymentRequest;
11  import com.example.restapi.dto.ReceiptResponse;
12  import com.example.restapi.service.CartService;
13  
14  @RestController
15  @RequestMapping("/api/carts")
16  public class CartController {
17  
18      private final CartService cartService;
19  
20      public CartController(CartService cartService) {
21          this.cartService = cartService;
22      }
23  
24      @GetMapping("/{ownerId}")
25      public ResponseEntity<CartResponse> getCart(@PathVariable String ownerId) {
26          UUID owner = UUID.fromString(ownerId);
27          CartResponse resp = cartService.getCart(owner);
28          return ResponseEntity.ok(resp);
29      }
30  
31      @PostMapping("/{ownerId}/items")
32      public ResponseEntity<CartResponse> addItem(@PathVariable String ownerId, @RequestBody CartRequest request) {
33          UUID owner = UUID.fromString(ownerId);
34          CartResponse resp = cartService.addItemToCart(owner, request.getItemId(), request.getQuantity());
35          return ResponseEntity.ok(resp);
36      }
37  
38      @DeleteMapping("/{ownerId}/items/{itemId}")
39      public ResponseEntity<CartResponse> removeItem(@PathVariable String ownerId, @PathVariable Long itemId) {
40          UUID owner = UUID.fromString(ownerId);
41          CartResponse resp = cartService.removeItemFromCart(owner, itemId);
42          return ResponseEntity.ok(resp);
43      }
44  
45      @PostMapping("/{ownerId}/clear")
46      public ResponseEntity<CartResponse> clearCart(@PathVariable String ownerId) {
47          UUID owner = UUID.fromString(ownerId);
48          CartResponse resp = cartService.clearCart(owner);
49          return ResponseEntity.ok(resp);
50      }
51  
52      @PostMapping("/{ownerId}/checkout")
53      public ResponseEntity<ReceiptResponse> checkout(@PathVariable String ownerId, @RequestBody PaymentRequest paymentRequest) {
54          UUID owner = UUID.fromString(ownerId);
55          ReceiptResponse receipt = cartService.checkout(owner, paymentRequest);
56          return ResponseEntity.ok(receipt);
57      }
58  }
