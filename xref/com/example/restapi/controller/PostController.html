1   package com.example.restapi.controller;
2   
3   
4   import com.example.restapi.dto.PostRequest;
5   import com.example.restapi.dto.PostResponse;
6   import com.example.restapi.model.Category;
7   import com.example.restapi.model.Post;
8   import com.example.restapi.model.Profile;
9   import com.example.restapi.service.PostService;
10  import org.springframework.http.HttpStatus;
11  import org.springframework.http.ResponseEntity;
12  import org.springframework.web.bind.annotation.*;
13  
14  import java.util.List;
15  import java.util.Map;
16  import java.util.UUID;
17  
18  @RestController
19  @RequestMapping("/api/posts")
20  public class PostController {
21  
22      private final PostService postService;
23  
24      public PostController(PostService postService) {
25          this.postService = postService;
26      }
27  
28      @GetMapping
29      public ResponseEntity<List<PostResponse>> getAllPosts() {
30          return ResponseEntity.ok(postService.getAllPosts());
31      }
32  
33      @GetMapping("/{id}")
34      public ResponseEntity<PostResponse> getPostById(@PathVariable Long id) {
35          return ResponseEntity.ok(postService.getPostById(id));
36      }
37  
38      @GetMapping("/by-author/{authorId}")
39      public ResponseEntity<List<PostResponse>> getPostsByAuthor(@PathVariable UUID authorId) {
40          return ResponseEntity.ok(postService.getPostsByAuthor(authorId));
41      }
42  
43      @PostMapping
44      public ResponseEntity<?> createPost(@RequestBody PostRequest req) {
45          try {
46              Post post = new Post();
47              post.setTitle(req.getTitle());
48              post.setContent(req.getContent());
49              if (req.getAuthorId() != null) {
50                  Profile author = new Profile();
51                  author.setId(req.getAuthorId());
52                  post.setAuthor(author);
53              }
54              if (req.getCategoryId() != null) {
55                  Category category = new Category();
56                  category.setId(req.getCategoryId());
57                  post.setCategory(category);
58              }
59              PostResponse created = postService.createPost(post);
60              return new ResponseEntity<>(created, HttpStatus.CREATED);
61          } catch (IllegalArgumentException e) {
62              return ResponseEntity.badRequest().body(e.getMessage());
63          }
64      }
65  
66      @PatchMapping("/{id}")
67      public ResponseEntity<PostResponse> updatePost(@RequestBody Map<String, Object> updates, @PathVariable Long id) {
68          try {
69              return ResponseEntity.ok(postService.updatePost(id, updates));
70          } catch (RuntimeException e) {
71              return ResponseEntity.notFound().build();
72          }
73      }
74  
75      @DeleteMapping("/{id}")
76      public ResponseEntity<Void> deletePost(@PathVariable Long id) {
77          postService.deletePost(id);
78          return new ResponseEntity<>(HttpStatus.NO_CONTENT);
79      }
80  }
